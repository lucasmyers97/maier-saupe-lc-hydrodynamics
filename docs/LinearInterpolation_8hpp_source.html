142 template <int dim>
143 void LinearInterpolation<dim>::vector_value_list(
144  const std::vector<dealii::Point<dim>> &points,
145  std::vector<dealii::Vector<double>> &vals) const
146 {
147  // check sizing
148  assert(points.size() == vals.size()
149  && "Number of points and values is different");
150  assert(vals[0].size() == f.size()
151  && "Input vector and sampled vector are different sizes");
152 
153  // find lower-left indices associated with each point
154  // allocate n_pts x dim indices
155  std::vector<std::vector<int>> idx(points.size(),
156  std::vector<int>(dim));
157  for (unsigned int pt_idx = 0; pt_idx < points.size(); ++pt_idx)
158  {
159  double num_x_gridpoints = (points[pt_idx][0] - x_start) / dx;
160  double num_y_gridpoints = (points[pt_idx][1] - y_start) / dy;
161  idx[pt_idx][0] = static_cast<unsigned int>( std::floor(num_x_gridpoints) );
162  idx[pt_idx][1] = static_cast<unsigned int>( std::floor(num_y_gridpoints) );
163  }
164 
165  // do the interpolation -- for an explanation of the coefficients, see wiki
166  for (unsigned int pt_idx = 0; pt_idx < points.size(); ++pt_idx)
167  {
168  // lower left grid indices corresponding to given point
169  unsigned int i = idx[pt_idx][0];
170  unsigned int j = idx[pt_idx][1];
171 
172  if (i < 0 || i >= (m - 1) || j < 0 || j >= (n - 1))
173  {
174  throw std::runtime_error(std::string(
175  "Error: Interpolation point is out of bounds\n"));
176  return;
177  }
178 
179  double x = points[pt_idx][0];
180  double y = points[pt_idx][1];
181  double x1 = dx * i + x_start;
182  double x2 = dx * (i + 1) + x_start;
183  double y1 = dy * j + y_start;
184  double y2 = dy * (j + 1) + y_start;
185 
186  // coefficients of final interpolation sum
187  double c11 = (x2 - x)*(y2 - y);
188  double c21 = (x - x1)*(y2 - y);
189  double c12 = (x2 - x)*(y - y1);
190  double c22 = (x - x1)*(y - y1);
191  double A = 1 / (dx*dy);
192 
193  for (unsigned int vec_idx = 0; vec_idx < vec_dim; ++vec_idx)
194  {
195  double f11 = f[vec_idx] [i][j];
196  double f21 = f[vec_idx] [i + 1][j];
197  double f12 = f[vec_idx] [i][j + 1];
198  double f22 = f[vec_idx] [i + 1][j + 1];
199  vals[pt_idx][vec_idx] = A * (f11*c11 + f21*c21 + f12*c12 + f22*c22);
200  }
201  }
202 }
