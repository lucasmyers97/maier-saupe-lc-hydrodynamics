181  // Find max elements in each row for scaling
182  for (i = 0; i < N; ++i)
183  {
184  max_element = 0;
185  for (j = 0; j < N; ++j)
186  {
187  if (A[N*i + j] > max_element)

189  }
190  row_scale[i] = max_element;
191  }
192 
193  for (k = 0; k < N; ++k)
194  {
195  // Find max element below k in current column
196  max_element = A[N*k + k];
197  imax = k;
198  for (i = k + 1; i < N; ++i)
199  {
200  if (A[N*i + k] / row_scale[i] > max_element)
201  {
202  max_element = A[N*i + k];
203  imax = i;
204  }
205  }
206 
207  // Promote max_element row, demote k-row
208  if (imax != k)
209  {
210  for (j = 0; j < N; ++j)
211  {
212  temp_element = A[N*k + j];
213  A[N*k + j] = A[N*imax + j];
214  A[N*imax + j] = temp_element;
215  }
216  // won't use <= k scale factors again, don't need to switch
217  row_scale[imax] = row_scale[k];
218  }
219  // Record in row_index array
220  row_index[k] = imax;
221 
222  // finish i > k, j = k elements
223  // add another term to i > k, j > k elements
224  for (i = k + 1; i < N; ++i)
225  {
226  // divide by pivot elements in column k for i > k
227  A[N*i + k] /= A[N*k + k];
