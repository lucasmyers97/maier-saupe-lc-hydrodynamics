307 LagrangeMultiplierGPU<T, order, vec_dim>::calcLambda(T* Q_in)
308 {
309  // Get initial residual, jacobian for given Q
310  initializeInversion(Q_in);
311 
312  // Calculate norm of initial residual
313  double res_norm = 0;
314  for (int i = 0; i < vec_dim; ++i)
315  res_norm += Res[i]*Res[i];
316  res_norm = sqrt(res_norm);
317 
318  // Do Newton's method until norm < tolerance, or iterations > max iterations
319  int iters = 0;
320  while (iters < max_iters && res_norm > tol)
321  {
322  calcdLambda();
323  for (int i = 0; i < vec_dim; ++i)
324  Lambda[i] += dLambda[i];
325 
326  calcResJac();
327 
328  res_norm = 0;
329  for (int i = 0; i < vec_dim; ++i)
330  res_norm += Res[i]*Res[i];
331  res_norm = sqrt(res_norm);
332 
333  ++iters;
334  }
335 
336  // Throw an error if inversion failed
337  assert(res_norm < tol);
338 
339  // Copy inversion back to global device memory
340  for (int i = 0; i < vec_dim; ++i)
341  Q_in[i] = Lambda[i];
342 }

385 __device__ inline
386 void
387 LagrangeMultiplierGPU<T, order, vec_dim>::calcResJac()
388 {
389  double exp_lambda{};
390  int row_idx{};
391 
392  // Arrays holding all values of Lebedev quadrature
393  double Z{};
394  double int1[vec_dim] = {0};
395  double int2[vec_dim*vec_dim] = {0};
396  double int3[vec_dim*vec_dim] = {0};
397  double int4[vec_dim] = {0};
398 
399  // Row & Col indices of Q-tensor corresponding to Q-vector entries
400  int i[vec_dim] = {0, 0, 0, 1, 1};
401  int j[vec_dim] = {0, 1, 2, 1, 2};
402 
403  // KroneckerDelta but in Q-vector form
404  double delta[vec_dim] = {1, 0, 0, 1, 0};
405 
406  // Calculate each term in Lebedev quadrature for each integral, add to total
407  // quadrature value until we've summed all terms
408  #pragma unroll
409  for (int coord_idx = 0; coord_idx < order; ++coord_idx)
410  {
411  row_idx = N_COORDS*coord_idx;
412  exp_lambda = calcExpLambda(row_idx);
413 
414  Z += exp_lambda * lebedev_weights[coord_idx];
415 
416  #pragma unroll
417  for (int m = 0; m < vec_dim; ++m)
418  {
419  int1[m] += calcInt1Term(exp_lambda, coord_idx, row_idx, i[m], j[m]);
420  int4[m] += calcInt4Term(exp_lambda, coord_idx, row_idx, i[m], j[m]);
421 
422  #pragma unroll
423  for (int n = 0; n < vec_dim; ++n)
424  {
425  int2[vec_dim*m + n] += calcInt2Term(exp_lambda, coord_idx,
426  row_idx, i[m], j[m],
427  i[n], j[n]);
428  int3[vec_dim*m + n] += calcInt3Term(exp_lambda, coord_idx,
429  row_idx, i[m], j[m],
430  i[n], j[n]);
431  }
432  }
433  }
434 
435  // Calculate each entry of residual and Jacobian using integral values
436  #pragma unroll
437  for (int m = 0; m < vec_dim; ++m)
438  {
439  Res[m] = int1[m] / Z
440  - (1.0 / 3.0) * delta[m]
441  - Q[m];
442 
443  #pragma unroll
444  for (int n = 0; n < vec_dim; ++n)
445  {
446  if (n == 0 || n == 3)
447  Jac(m, n) = int3[vec_dim*m + n] / Z
448  - int1[m] * int4[n] / (Z*Z);
449  else
450  Jac(m, n) = 2 * int2[vec_dim*m + n] / Z
451  - 2 * int1[m] * int1[n] / (Z*Z);
452  }
453  }
454 }
